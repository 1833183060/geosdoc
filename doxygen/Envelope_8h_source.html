139 static bool intersects(
140 const Coordinate& p1, const Coordinate& p2,
141 const Coordinate& q1, const Coordinate& q2)
142 {
143 double minq = std::min(q1.x, q2.x);
144 double maxq = std::max(q1.x, q2.x);
145 double minp = std::min(p1.x, p2.x);
146 double maxp = std::max(p1.x, p2.x);
147 if(minp > maxq) {
148 return false;
149 }
150 if(maxp < minq) {
151 return false;
152 }
153 minq = std::min(q1.y, q2.y);
154 maxq = std::max(q1.y, q2.y);
155 minp = std::min(p1.y, p2.y);
156 maxp = std::max(p1.y, p2.y);
157 if(minp > maxq) {
158 return false;
159 }
160 if(maxp < minq) {
161 return false;
162 }
163 return true;
164 }

667 double distanceSquared(const Envelope& env) const
668 {
669 double dx = std::max(0.0,
670 std::max(maxx, env.maxx) - std::min(minx, env.minx) - (maxx - minx) -
671 (env.maxx - env.minx));
672 double dy = std::max(0.0,
673 std::max(maxy, env.maxy) - std::min(miny, env.miny) - (maxy - miny) -
674 (env.maxy - env.miny));
675
676 return dx * dx + dy * dy;
677 };

705 static double distanceSquaredToCoordinate(
706 const Coordinate& c,
707 const Coordinate& p0,
708 const Coordinate& p1)
709 {
710 double xa = c.x - p0.x;
711 double xb = c.x - p1.x;
712 double ya = c.y - p0.y;
713 double yb = c.y - p1.y;
714
715 // If sign of a and b are not the same, then Envelope spans c and distance is zero.
716 double dx = (std::signbit(xa) == std::signbit(xb)) * std::min(std::abs(xa), std::abs(xb));
717 double dy = (std::signbit(ya) == std::signbit(yb)) * std::min(std::abs(ya), std::abs(yb));
718
719 return dx*dx + dy*dy;
720 }
